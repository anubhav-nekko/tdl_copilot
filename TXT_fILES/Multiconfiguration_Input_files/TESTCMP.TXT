; Purpose: Sample TDL report for Tally, displaying summarized item-wise
;          quantities and values from voucher inventory entries.
;===============================================================================

;------------------------------------------------------------------------------
; MENU INTEGRATION
;------------------------------------------------------------------------------

[#menu: Gateway of Tally]
    add: Option: textaaLock ;; : @@textaaDemoLock

[!menu: textaaLock]
    add: Item: before: @@locQuit: @@textaaReport: Display: Reptextaa
    add: Item: before: @@locQuit: Blank

[System: formula]
    textaaReport: "textaa"
;; textaaDemoLock: $$MachineDate < $$Date:"01/04/2013"

;------------------------------------------------------------------------------
; MAIN REPORT AND FORM
;------------------------------------------------------------------------------

[Report: Reptextaa]
    use: Dsp Template
    Title: @@textaaReport
    Printset: Report Title: @@textaaReport
    Form: Frmtextaa
    Export: Yes
    set: svfromdate: ##svcurrentdate
    set: svTodate: ##svcurrentdate
    Local: Button: RelReports: Inactive: Yes

[Form: Frmtextaa]
    use: DSP Template
    Part: DspAccTitles, PrtTitle0textaa, Prttextaa
    Width: 100% Page
    Height: 100% Page
    Background: @@SV_STOCKSUMMARY
    delete: page break
    add: page break: textaabotbrk, textaabotOpbrk
    Bottom Toolbar Buttons: BottomToolBarBtn1, BottomToolBarBtn3, BottomToolBarBtn8, BottomToolBarBtn9, BottomToolBarBtn10, BottomToolBarBtn11, BottomToolBarBtn12

[part: textaabotBrk]
    line: EXPINV PageBreak
    border: thin top

[part: textaabotopbrk]
    use: dspacctitles
    add: part: textaaTitlePart

[part: textaaTitlePart]
    line: LntextaaTitle

[line: LntextaaCurrPeriod]
    field: fwf, fwf2
    Local: field: fwf2: Align: Right
    Local: Field: fwf: Style: normal bold
    Local: Field: fwf2: Style: normal bold
    Local: Field: fwf2: Set As: @@dspDateStr
    invisible: $$inprintmode

[part: PrtTitle0textaa]
    line: LntextaaCurrPeriod

[Part: Prttextaa]
    Line: LntextaaTitle, Lntextaa
    bottom Line: LntextaaTotals
    repeat: Lntextaa: Coltextaa
    scroll: Vertical
    Common Border: Yes
    Total: Qtyf, amtf

;------------------------------------------------------------------------------
; DATA COLLECTION: ITEM-WISE AGGREGATION FROM INVENTORY ENTRIES
;------------------------------------------------------------------------------

[Collection: Coltextaa]
    source Collection: default ;;Coltextaasrc
    walk: inventoryentries
    by: stockitemname: $stockitemname
    aggr compute: billedqty: sum: $billedqty
    by: rate: $rate
    aggr compute: amount: sum: $amount

[Collection: Coltextaasrc]
    Use: Vouchers of Company
    delete: filter: daybookfilter
    Filter: ColtextaaFilter, IsNonOptionalCancelledVchs

[system: Formula]
    ColtextaaFilter: Yes

;------------------------------------------------------------------------------
; COLUMN HEADERS
;------------------------------------------------------------------------------

[Line: LntextaaTitle]
    use: Lntextaa
    option: titleopt
    local: field: sdf: set as: "Date"
    local: field: nf: set as: "Name"
    local: field: fwf: set as: "Description"
    local: field: qtyf: set as: "Qty."
    local: field: amtf: set as: "Value"
    local: field: ratepf: set as: "Rate"
    local: field: default: style: normal bold

;------------------------------------------------------------------------------
; MAIN DATA LINE
;------------------------------------------------------------------------------

[Line: Lntextaa]
    Fields: sdf, nf, fwf
    right field: numf, Qtyf, Amtf
    Option: Alter on Enter
    local: field: qtyf: Format: "NoSymbol, Short Form, No Compact,NoZero"
    local: field: ratepf: setas: #amtf/#qtyf
    local: field: fwf: alter: voucher: $$isvoucher
    option: alter on enter
    local: field: fwf: alter: voucher: $$isvoucher
    local: field: nf: set as: $stockitemname
    Local: Field: numf: Set As: $rate
    local: field: sdf: set as: $date

;------------------------------------------------------------------------------
; TOTALS LINE
;------------------------------------------------------------------------------

[line: LntextaaTotals]
    use: Lntextaa
    option: totalOpt
    local: field: fwf: align: right
    local: field: default: style: normal bold
    local: field: qtyf: set as: $$total:qtyf
    local: field: fwf: set as: "Total"
    local: field: fwf: set as: ""
    local: field: amtf: set as: $$total:amtf
