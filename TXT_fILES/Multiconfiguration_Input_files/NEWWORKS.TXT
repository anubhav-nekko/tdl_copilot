; AUTO-FILL ACTUAL & BILLED QTY IN BATCH ALLOCATIONS FOR SALES/DEL NOTE
;------------------------------------------------------------------------------

[#Line: STKVCH Batch2] ;;$date < $$date:"02/12/2011" and 
    add : option : s2 : $$increatemode and (@@isdelnote or @@issales and $$issysname:#VCHBATCHTrack)

[!line : s2]
    local:field:VCHBATCH ActualQty: set as : #vchbatchbilledqty
    local:field:VCHBATCH BilledQty: set as : if $$increatemode AND $$ISEMPTY:$$vALUE then $$asqty:1 else $$value
    local:field:VCHBATCH ActualQty: set always : yes
