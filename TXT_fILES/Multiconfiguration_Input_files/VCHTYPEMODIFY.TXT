; Part definition for barcode auto-invoke functionality
[#PART : CWBARCODEAUTOINVOKE]

; Adds a new line after the warning for empty party code
add:line:after:cwWarnonEmptypartyCode:cwcomitemline

; Line definition for combined item configuration
[line:cwcomitemline]
 field: long prompt,cwlogical  ; Defines fields for the line: a prompt and a logical field
  Local: Field: long prompt: info: "Combined Stock Item in Print ?"  ; Sets the prompt text for the option
  Local: Field: long prompt: width:30  ; Sets the width of the prompt field to 30 units
   Local: Field: cwlogical: storage:cwcomitem  ; Links the logical field to the cwcomitem storage variable

  Local: Field: default: Style: Normal Bold  ; Sets the default style for fields to bold
  
  ; Formula to retrieve the combined item option setting for the current voucher type
  [System: Formula]
  cwcomitemopt:$cwcomitem:vouchertype:$vouchertypename
