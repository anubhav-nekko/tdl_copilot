; ==================== BARCODE AUTO INVOKE PART DEFINITION ======================
; This section defines the configuration for automatic barcode invocation functionality

[#PART : CWBARCODEAUTOINVOKE]
; Adds a new line after the warning on empty party code line
add:line:after:cwWarnonEmptypartyCode:cwcomitemline

; ==================== COMBINED STOCK ITEM CONFIGURATION ========================
; Configuration for combined stock item printing option in barcode module

[line:cwcomitemline]
 field: long prompt,cwlogical
  ; Label for the combined stock item printing option
  Local: Field: long prompt: info: "Combined Stock Item in Print ?"
  ; Sets the width of the prompt field
  Local: Field: long prompt: width:30
  ; Storage field for the combined item setting
   Local: Field: cwlogical: storage:cwcomitem

  ; Style setting for the field
  Local: Field: default: Style: Normal Bold
  
  ; ==================== SYSTEM FORMULA FOR VOUCHER TYPE SETTING ===============
  ; Formula to retrieve the combined item setting for the current voucher type
  [System: Formula]
  cwcomitemopt:$cwcomitem:vouchertype:$vouchertypename
