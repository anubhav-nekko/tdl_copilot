;          bill collection, filtering, and reporting with advanced grouping,
;          period setup, and professional formatting.

;;------------------------------------------------------------------------------
;; MENU INTEGRATION: Add report option to Gateway of Tally and Debug menu
;;------------------------------------------------------------------------------

[#menu: Gateway of Tally]
    ;; Optionally lock demo access by date if needed
    ;; add: Option: tes222Lock ;; : @@tes222DemoLock

[#menu : cw_Debug_menu]
    ;; Add the report to the debug menu for quick access
    ;; add: Item: before: @@locQuit: @@tes222Report: Display: Reptes222

[!menu: tes222Lock]
    add: Item: before: @@locQuit: @@tes222Report: Display: Reptes222
    add: Item: before: @@locQuit: Blank

[System: formula]
    tes222Report: "tes222"
;; tes222DemoLock: $$MachineDate < $$Date:"01/04/2013"

;;------------------------------------------------------------------------------
;; MAIN REPORT DEFINITION
;;------------------------------------------------------------------------------

[Report: Reptes222]
    use        : Dsp Template
    Title      : @@tes222Report
    Printset   : Report Title: @@tes222Report
    Form       : Frmtes222
    Export     : Yes
    set        : svfromdate : ##svcurrentdate
    set        : svTodate   : ##svcurrentdate
    Local      : Button   : RelReports : Inactive : Yes

;;------------------------------------------------------------------------------
;; MAIN FORM LAYOUT
;;------------------------------------------------------------------------------

[Form: Frmtes222]
    use     : DSP Template
    Part    : DspAccTitles, PrtTitle0tes222, Prttes222
    Width   : 100% Page
    Height  : 100% Page
    ;; Background: @@SV_STOCKSUMMARY
    delete  : page break
    add     : page break: tes222botbrk, tes222botOpbrk
    Bottom Toolbar Buttons: BottomToolBarBtn1, BottomToolBarBtn3, BottomToolBarBtn8, BottomToolBarBtn9, BottomToolBarBtn10, BottomToolBarBtn11, BottomToolBarBtn12
    ;; Standard Tally navigation and action buttons

[part: tes222botBrk]
    line: EXPINV PageBreak
    border: thin top

[part: tes222botopbrk]
    use: dspacctitles
    add: part: tes222TitlePart

[part: tes222TitlePart]
    line: Lntes222Title

[line: Lntes222CurrPeriod]
    field: fwf,fwf2
    Local: field: fwf2: Align: Right
    Local: Field: fwf: Style: normal bold
    Local: Field: fwf2: Style: normal bold
    Local: Field: fwf2: Set As: @@dspDateStr
    invisible: $$inprintmode

[part: PrtTitle0tes222]
    line : Lntes222CurrPeriod

;;------------------------------------------------------------------------------
;; MAIN DATA PART: Table of bills and grouped details
;;------------------------------------------------------------------------------

[Part: Prttes222]
    Line: Lntes222Title, Lntes222
    bottom Line: Lntes222Totals
    repeat: Lntes222: Coltes222
    scroll: Vertical
    Common Border: Yes
    Total: Qtyf, amtf

;;------------------------------------------------------------------------------
;; DATA COLLECTION: Bill details, grouped by parent, with cleared/uncleared split
;;------------------------------------------------------------------------------

[Collection: Coltes222]
    Collection: Coltes222a
    Collection: Coltes222b

[Collection: Coltes222a]
    Type : Bills
    fetch : cwcaption1vch
    fetch : LedgerEntries.InventoryEntries.BilledQty
    fetch : LedgerEntries.vouchernumber
    fetch : LedgerEntries.vouchertypename
    fetch : BillType
    ;; fetch : LedgerEntries.LedgerEntries.vouchertypename
    by:parent:$parent
    ;; filter:Coltes222Filter
    cleared : yes
    compute : cwSalesDuringThePeriodQty :@@cwsalesqtycns

[Collection: Coltes222b]
    use:Coltes222a
    cleared : no

[system: Formula]
    ;; Example filter for specific parent names (edit as needed)
    ;; Coltes222Filter:$cwcaption1item:ledger:$parent="CASH PARTY,DIRECT"
    ;; Coltes222Filter:$cwcaption1item:ledger:$parent="DIRECT,KED (OTHERS)"
    Coltes222Filter:$parent="NAURATAN AGARWAL,NBISL/NDIN"

;;------------------------------------------------------------------------------
;; HEADER LINE: Column titles for the report
;;------------------------------------------------------------------------------

[Line: Lntes222Title]
    use: Lntes222
    option: titleopt
    local:field: sdf: set as: "Date"
    local:field: nf: set as: "Name"
    local:field: fwf: set as: "Description"
    local:field: qtyf: set as: "Qty."
    local:field: amtf: set as: "Value"
    local:field: ratepf : set as : "Rate"
    local: field: default : style: normal bold

;;------------------------------------------------------------------------------
;; DATA LINE: Main line showing all calculated columns per bill
;;------------------------------------------------------------------------------

[Line: Lntes222]
    Fields: nf1, sdf, snf, nf, fwf
    right field: ratepf, numf, Amtf
    Option: Alter on Enter
    local:field: qtyf : Format : "NoSymbol, Short Form, No Compact,NoZero"
    local:field: ratepf : setas  : #amtf/#qtyf
    local: field: fwf: alter : voucher : $$isvoucher
    option : alter on enter
    local : field : fwf : alter : voucher : $$isvoucher
    ;; local : field : fwf : alter : ledger : $$isledger
    local : field : fwf : set as : $parent
    local : field : sdf : set as : $billdate
    local : field : snf : set as : $name
    local : field : nf1 : set as : $$CollectionField:$vouchertypename:First:LedgerEntries
    local : field : nf : set as : $$CollectionField:$vouchernumber:First:LedgerEntries
    ;; local : field : numf : set as :@@cwsalesqtycns  ;;$cwSalesDuringThePeriodQty
    local : field : numf : set as :@@cwsalesqtycns

;;------------------------------------------------------------------------------
;; TOTALS LINE: Running totals for all columns
;;------------------------------------------------------------------------------

[line: Lntes222Totals]
    use: Lntes222
    option: totalOpt
    local: field: fwf: align: right
    local: field: default : style: normal bold
    local: field: qtyf: set as: $$total:qtyf
    local: field: fwf: set as: "Total"
    local: field: fwf: set as: ""
    local: field: amtf : set as :  $$total:amtf
