;===============================================================================
; STOCKREPORT.TXT
; Created By: Khokan on 2021-09-30 17:32, ID:
; Purpose: Implements a "Stock Report" in Tally, listing all stock items with
;          their description and unit, supporting filtering by item name,
;          running totals, and professional formatting for export/print.
;===============================================================================

;;------------------------------------------------------------------------------
;; MENU INTEGRATION: Add report option to Gateway of Tally and Debug menu
;;------------------------------------------------------------------------------

[#menu: Gateway of Tally]
    ;; Optionally lock demo access by date if needed
    ;; add: Option: StockreportLock ;; : @@StockreportDemoLock

[#menu : cw_Debug_menu]
    add: Item: before: @@locQuit: @@StockreportReport: Display: RepStockreport

[!menu: StockreportLock]
    add: Item: before: @@locQuit: @@StockreportReport: Display: RepStockreport
    add: Item: before: @@locQuit: Blank

[System: formula]
    StockreportReport: "Stock report"
;; StockreportDemoLock: $$MachineDate < $$Date:"01/04/2013"

;;------------------------------------------------------------------------------
;; MAIN REPORT DEFINITION
;;------------------------------------------------------------------------------

[Report: RepStockreport]
    use        : Dsp Template
    Title      : @@StockreportReport
    Printset   : Report Title: @@StockreportReport
    Form       : FrmStockreport
    Export     : Yes
    set        : svfromdate : ##svcurrentdate
    set        : svTodate   : ##svcurrentdate
    Local      : Button   : RelReports : Inactive : Yes
    variable   : stockitemname

;;    set:stockitemname:"SAREE TWENTY TWENTY (BAG)"

;;------------------------------------------------------------------------------
;; MAIN FORM LAYOUT
;;------------------------------------------------------------------------------

[Form: FrmStockreport]
    use     : DSP Template
    Part    : DspAccTitles,PrtTitle0Stockreport,PrtStockreport
    Width   : 100% Page
    Height  : 100% Page
    Background: @@SV_STOCKSUMMARY
    delete  : page break
    add     : page break: Stockreportbotbrk,StockreportbotOpbrk
    Bottom Toolbar Buttons 	: BottomToolBarBtn1, BottomToolBarBtn3, BottomToolBarBtn8, BottomToolBarBtn9, BottomToolBarBtn10, BottomToolBarBtn11,BottomToolBarBtn12
    ;; Style company name, address, and report title for print
    local:Part:DSPCompanyName:local:line:DSPCompanyName: Local: Field: DSP CompanyName:PrintStyle:styleCalisto2
    local:Part:DSPCompanyAddress:local:line:DSPCompanyAddress: Local: Field: DSPCompanyAddress:PrintStyle:style2n
    local:Part:DSPReportTitle:local:line:DSPReportName: Local: Field: DSPReportName:PrintStyle:style3n
    local:Part:DSPReportSubTitle:local:line:DSPReportSubTitle: Local: Field: DSPReportSubTitle:PrintStyle:style2n
    local:Part:DSPReportTitle:local:line:DSPReportPeriod:Local: Field: DSPReportPeriod:PrintStyle:style2n
    local:Part:DSPPageNo:local:line:DSPPageNo: Local: Field:DSPPageNo:PrintStyle:style2n

[part: StockreportbotBrk]
    line: EXPINV PageBreak
    border: thin top

[part: Stockreportbotopbrk]
    use: dspacctitles
    add: part: StockreportTitlePart

[part: StockreportTitlePart]
    line: LnStockreportTitle

[line: LnStockreportCurrPeriod]
    field: fwf,fwf2
    Local: field: fwf2: Align: Right
    Local: Field: fwf: Style: normal bold
    Local: Field: fwf2: Style: normal bold
    Local: Field: fwf2: Set As: @@dspDateStr
    invisible: $$inprintmode

[part: PrtTitle0Stockreport]
    line : LnStockreportCurrPeriod

;;------------------------------------------------------------------------------
;; MAIN DATA PART: Table of stock items
;;------------------------------------------------------------------------------

[Part: PrtStockreport]
    Line: LnStockreportTitle,LnStockreport
    bottom Line: LnStockreportTotals
    repeat: LnStockreport: ColStockreport
    scroll: Vertical
    Common Border: YEs
    Total: Qtyf,amtf

[Collection: ColStockreport]
    type : stockitem
    Filter: cwstkitemFilter

[system: Formula]
    cwstkitemFilter:if $$issysname:##stockitemname then yes else $name=##stockitemname

[system: Formula]
    ColStockreportFilter: Yes

;;------------------------------------------------------------------------------
;; HEADER LINE: Column titles for the report
;;------------------------------------------------------------------------------

[Line: LnStockreportTitle]
    use: LnStockreport
    option: titleopt
    local:field: fwf: set as: "Description"
    local:field: snf: set as: "unit"
    local: field: default : style: normal bold

;;------------------------------------------------------------------------------
;; DATA LINE: Main line showing all stock items
;;------------------------------------------------------------------------------

[Line: LnStockreport]
    Fields:fwf,d1
    right field:snf,d2
    Option: Alter on Enter
    local:field: qtyf : Format : "NoSymbol, Short Form, No Compact,NoZero"
    local:field: ratepf : setas  : #amtf/#qtyf
    local: field: fwf: alter : voucher : $$isvoucher
    option : alter on enter
    local : field : fwf : alter : voucher : $$isvoucher
    Local: Field: fwf: Set As:$name
    Local: Field: snf: Set As:$baseunits

;;------------------------------------------------------------------------------
;; TOTALS LINE: Running totals for all columns
;;------------------------------------------------------------------------------

[line: LnStockreportTotals]
    use: LnStockreport
    option: totalOpt
    local: field: fwf: align: right
    local: field: default : style: normal bold
    local: field: qtyf: set as: $$total:qtyf
    local: field: fwf: set as: "Total"
    local: field: fwf: set as: ""
    local: field: amtf : set as :  $$total:amtf

;===============================================================================
; End of STOCKREPORT.TXT (with documentation comments)
;===============================================================================
