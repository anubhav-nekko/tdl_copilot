;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Report Definition
;; Binds the Report `RepSimpleFormat` to a specific Voucher object using ##myid
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

[report :RepSimpleFormat]
Object : Voucher : ##myid      ; The report fetches data from the Voucher identified by ##myid
form:RepSimpleFormat           ; Form to be used for rendering the report

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Main Form Definition for Simple Invoice Format
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

[form : RepSimpleFormat]
Top part     : RepSimpleFormatTit, cwBajCmpDetailsPart, cwBajPartyDetailsPart  ; Header sections
Part         : cwSimpleFormatInvACCPart                                        ; Main content (Inventory + Accounting)
bottom Part  : RepBajajExportPdfBottomPart                                    ; Footer section
page Break   : cwSimpleFormatCLBreak, cwSimpleFormatOPBreak                   ; Page breaks at designated parts

;; Setting local value for `amtf` field in totals line (footer)
local:part:RepBajajExportPdfBottomPart:local:line:RepBajajExportPdfTotLine:Local: Field: amtf: Set As:$amount

;; Add border to party details part
local:part:cwBajPartyDetailsPart:border:thin left right bottom

;; Dimensions and spacing of the printed page
Width         : @@InvWidth Inch
Height        : @@InvHeight Inch
Space Top     : @@InvSpace inch
Space Right   : 0.5 inch
Space Left    : @@InvSpaceLeft inch
Space Bottom  : 0.25 inch

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Title Part of the Invoice
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

[part : RepSimpleFormatTit]
line:RepSimpleFormatTit

[line: RepSimpleFormatTit]
field:fwfc

;; Set the field value as Voucher Print Title for the Voucher Type
Local: Field: fwfc: Set As:$VCHPrintTitle:vouchertype:$vouchertypename
Local: Field: fwfc: Border: thin Bottom
Local: Field: fwfc: Style: Large Bold

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Page Break Parts for Closing and Opening
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

[Part : cwSimpleFormatCLBreak]
use : RepBajajExportPdfBottomPart    ; Reuse footer part as page break component

[Part : cwSimpleFormatOpBreak]
part : RepSimpleFormatTit, cwBajCmpDetailsPart, cwBajPartyDetailsPart  ; Reprint header components

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Combined Inventory and Accounting Part
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

[part : cwSimpleFormatINVACCPart]
Parts : RepBajajExportPdfInvPart, RepBajajExportPdfAccPart  ; Include both inventory and accounts sections

;; Control visibility of serial number fields based on conditions
local:line:RepBajajExportPdfInvLineTitle:local: field: snf2: Invisible:not $cwdonoprint:vouchertype:$vouchertypename
local:line:RepBajajExportPdfInvLine:local: field: snf2: Invisible:not $cwdonoprint:vouchertype:$vouchertypename
local:line:RepBajajExportPdfInvLineTitle:local: field: snf3: Invisible:not $cwserialprint:vouchertype:$vouchertypename
local:line:RepBajajExportPdfInvLine:local: field: snf3: Invisible:not $cwserialprint:vouchertype:$vouchertypename

;; Layout properties
CommonBorder : Yes        ; Apply common border to all cells
Vertical     : Yes        ; Enable vertical display of parts
Scroll       : Vertical   ; Enable vertical scroll
Float        : No         ; Do not allow floating of this part
Total        : amtf       ; Define total field (for amount)
border       : thin left right bottom ; Border style for this section
