[System: Formula]

; Determines primary key based on whether separate partner ID option is enabled
cwsubprimarykeyforUpload : 
    if @@cwsperateparnewridenbopt 
    then $cwsubprimarykey:Vouchertype:##SVVoucherType  
    else $cwsubprimarykey:COMPANY:##SVCURRENTCOMPANY

; Derived field used in formula above
cwsperateparnewridenbopt : $cwSepratepatnerid:company:##svcurrentcompany
cwconfigsubkey : $cwsubprimarykey:company:##svcurrentcompany
cwconfigpartnerid : not $$isempty:$cwpartenerid:company:##svcurrentcompany
cwexportlocationpath : $cwexportlocationpath:company:##svcurrentcompany
cwDOUpdateFromReport : $cwDOUpdateFromReport:COMPANY:##SVCURRENTCOMPANY
cwExpPdfFromReport : $cwExpPdfFromReport:COMPANY:##SVCURRENTCOMPANY
cwBFLInvFormat : $cwBFLInvFormat:COMPANY:##SVCURRENTCOMPANY
